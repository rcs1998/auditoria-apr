rules_version = '2'; // ═══════════════════════════════════════════════════════════════ // REGRAS DE SEGURANÇA — Auditoria APR // Compatível com o plano gratuito (Spark) do Firebase: usa apenas // recursos nativos do Firestore (get() para leitura de outro doc), // sem Cloud Functions nem custom claims (que exigem Blaze). // // Níveis de acesso (campo "nivel" em /usuarios/{uid}): // - tecnico: cria auditorias e lê as próprias; lê unidades/perguntas // (somente leitura, sem alterar nada de configuração) // - gestor: tudo do técnico + lê/edita TODAS as auditorias, // gerencia perguntas e usuários // - admin: tudo do gestor + gerencia unidades e parceiros // ═══════════════════════════════════════════════════════════════ service cloud.firestore { match /databases/{database}/documents { // ─────────────────────────────────────────────── // Helpers // ─────────────────────────────────────────────── function estaLogado() { return request.auth != null; } function meuNivel() { return get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data.nivel; } function souGestorOuAdmin() { return estaLogado() && meuNivel() in ['gestor', 'admin']; } function souAdmin() { return estaLogado() && meuNivel() == 'admin'; } // Fallback legado: e-mails cadastrados em config/admins também // são tratados como admin, para compatibilidade com o sistema antigo. function souAdminLegado() { return estaLogado() && request.auth.token.email in get(/databases/$(database)/documents/config/admins).data.emails; } function ehAdminCompleto() { return souAdmin() || souAdminLegado(); } // ─────────────────────────────────────────────── // usuarios/{uid} — perfil e nível de acesso // ─────────────────────────────────────────────── match /usuarios/{uid} { // Qualquer usuário autenticado pode ler a própria lista de usuários // (necessário para o app verificar o próprio nível no login). allow read: if estaLogado(); // Um usuário pode criar o PRÓPRIO documento (primeiro login). allow create: if estaLogado() && request.auth.uid == uid; // O próprio usuário só pode alterar o campo "exigeTrocaSenha" do seu // documento (usado para marcar que já trocou a senha do primeiro acesso) // — não pode alterar seu próprio nível ou outros campos. allow update: if souGestorOuAdmin() || ehAdminCompleto() || (estaLogado() && request.auth.uid == uid && request.resource.data.diff(resource.data).affectedKeys() .hasOnly(['exigeTrocaSenha'])); allow delete: if ehAdminCompleto(); } // ─────────────────────────────────────────────── // config/admins — lista legada de e-mails admin (fallback) // ─────────────────────────────────────────────── match /config/admins { allow read: if estaLogado(); allow write: if ehAdminCompleto(); } // ─────────────────────────────────────────────── // config/perguntas — perguntas do formulário + nomes de seção // ─────────────────────────────────────────────── match /config/perguntas { // Técnico precisa ler para preencher o formulário. allow read: if estaLogado(); // Só gestor/admin editam as perguntas. allow write: if souGestorOuAdmin() || ehAdminCompleto(); } // ─────────────────────────────────────────────── // unidades/{id} e subcoleção parceiros/{id} // ─────────────────────────────────────────────── match /unidades/{unidadeId} { allow read: if estaLogado(); // Apenas admin gerencia unidades. allow write: if ehAdminCompleto(); match /parceiros/{parceiroId} { allow read: if estaLogado(); allow write: if ehAdminCompleto(); } } // ─────────────────────────────────────────────── // auditorias/{id} — registros de auditoria preenchidos // ─────────────────────────────────────────────── match /auditorias/{auditoriaId} { // Técnico só pode criar (enviar) novas auditorias e ler as // que ele mesmo criou. Gestor/Admin leem todas. allow create: if estaLogado() && request.resource.data.tecnico.uid == request.auth.uid; allow read: if souGestorOuAdmin() || ehAdminCompleto() || (estaLogado() && resource.data.tecnico.uid == request.auth.uid); // Edição/exclusão de auditorias só para gestor/admin // (ex.: corrigir um registro, remover duplicidade). allow update, delete: if souGestorOuAdmin() || ehAdminCompleto(); } } }
